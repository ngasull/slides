Fragments & Declarativness


function RespondFragment(props: Props) {
  const { respondSa, viewer } = props
  return [
    <RespondSaBar
      selectedRespondSaId={respondSa.id}
      viewer={viewer}
    />
    <RespondFeed
      filter={filter}
      respondSa={respondSa}
    />
    <RespondConversation
      filter={filter}
      respondSa={respondSa}
      viewer={viewer}
    />
  ]
}
  

fragment Respond_viewer on Viewer {
  ...RespondConversation_viewer
  ...RespondSaBar_viewer
}

fragment Respond_respondSa on RespondSocialAccount {
  id
  ...RespondConversation_respondSa
  ...RespondFeed_respondSa
}
